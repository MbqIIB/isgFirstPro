## ============= Linux MLI MQ setup ======================

#### === ACCOUNT ============


###receiver:
./crtmqm -q mli.acct.rcvr.qm
./strmqm mli.rcvr.acct.qm
./runmqsc mli.rcvr.acct.qm
define qlocal (ACCT_CLI_Q)
define listener (ACCT_CLI_LISTENER) trptype (tcp) control (qmgr) port(1470) ipaddr(yourMachineIP)
start listener (ACCT_CLI_LISTENER)
define channel(SYSTEM.ADMIN.SVRCONN) CHLTYPE(SVRCONN)
start channel(SYSTEM.ADMIN.SVRCONN)
define channel (ifrend.acct.sender) chltype(rcvr) trptype(tcp)
** disale the channel authentication for now, to enable intercommunication
alter qmgr chlauth(disabled)
end


#### iFREND-side sender:
crtmqm -q ifrend.sender.acct.qm
strmqm ifrend.sender.acct.qm
runmqsc
define qlocal (ACCT_CLI_Q) usage (xmitq)
define qremote (ACCT_CLI_QR) rname (ACCT_CLI_D) rqmname ('mli.acct.rcvr.qm') xmitq (ACCT_CLI_Q)


define listener (ACCT_CLI_LISTENER) trptype (tcp) control (qmgr) port (1470)
start listener (ACCT_CLI_LISTENER)
define channel(SYSTEM.ADMIN.SVRCONN) CHLTYPE(SVRCONN)
start channel(SYSTEM.ADMIN.SVRCONN)

define channel (ifrend.acct.sender) chltype (sdr) conname('192.168.11.14(1470)') xmitq (ACCT_CLI_Q) trptype(tcp)

**note: before starting the channel, define first the receiving channel at the receiver qm. 

start channel(ifrend.acct.sender)

** disale the channel authentication for now, to enable intercommunication
alter qmgr chlauth(disabled)

end
